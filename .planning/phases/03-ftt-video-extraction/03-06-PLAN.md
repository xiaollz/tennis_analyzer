   ```python
   video_files = sorted(Path('knowledge/extracted/ftt_videos').glob('*.json'))
   ```

   ```python
   stats = {"total_raw": 0, "new_added": 0, "deduped": 0, "per_video": {}}
   all_edges = []
   all_chains = []
   
   for vf in video_files:
       data = json.loads(vf.read_text())
       video_id = vf.stem
       video_new = 0
       video_dedup = 0
       
       for c_data in data.get("concepts", []):
           stats["total_raw"] += 1
           concept = Concept(**c_data)
           dup_id = registry.add(concept)
           if dup_id is None:
               video_new += 1
               stats["new_added"] += 1
           else:
               video_dedup += 1
               stats["deduped"] += 1
       
       all_edges.extend(data.get("edges", []))
       all_chains.extend(data.get("diagnostic_chains", []))
       stats["per_video"][video_id] = {"new": video_new, "deduped": video_dedup}
   ```
