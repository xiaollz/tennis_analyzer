class ConceptType(str, Enum):
    TECHNIQUE = "technique"
    BIOMECHANICS = "biomechanics"
    DRILL = "drill"
    SYMPTOM = "symptom"
    MENTAL_MODEL = "mental_model"
    CONNECTION = "connection"

class Concept(BaseModel):
    id: str
    name: str
    name_zh: str
    aliases: list[str]
    category: ConceptType
    sources: list[str]
    description: str
    vlm_features: list[str]
    muscles_involved: list[str]
    active_or_passive: str | None
    confidence: float

class Edge(BaseModel):
    source_id: str
    target_id: str
    relation: RelationType  # causes, prevents, requires, contradicts, supports, drills_for, visible_as
    confidence: float
    evidence: str
    source_file: str
    resolution: str | None

class DiagnosticChain(BaseModel):
    id: str  # dc_ prefix
    symptom: str
    symptom_zh: str
    symptom_concept_id: str
    check_sequence: list[DiagnosticStep]
    root_causes: list[str]
    drills: list[str]
    priority: int
    vlm_frame: str | None
```

class KnowledgeGraph:
    graph: nx.MultiDiGraph
    def add_concept(self, concept: Concept) -> None
    def add_edge(self, edge: Edge) -> None
    def get_causal_chain(self, symptom_id: str, cause_type: str = "causes") -> list[list[str]]
    def to_json(self, path: Path) -> None
    @classmethod
    def from_json(cls, path: Path) -> KnowledgeGraph
    @property
    def node_count(self) -> int
    @property
    def edge_count(self) -> int
```
